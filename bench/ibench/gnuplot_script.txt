set terminal png size 1920,1080
set output outfile

set title titlestr noenhanced
set xla 'query number'
set yla 'query latency (s)'

set logscale y

set style line 1 linecolor rgb '#0000ff' linetype 1 linewidth 0.25 pointtype 6 pointsize 0.25
set style line 2 linecolor rgb '#ff0000' linetype 1 linewidth 0.25 pointtype 6 pointsize 0.25
set style line 3 linecolor rgb '#00ff00' linetype 1 linewidth 0.25 pointtype 6 pointsize 0.25
set style line 4 linecolor rgb '#000000' linetype 1 linewidth 0.25 pointtype 6 pointsize 0.25

stats filename1
global_min = STATS_min
global_max = STATS_max

stats filename2
if (global_min < STATS_min) global_min = STATS_min
if (global_max > STATS_max) global_max = STATS_max

stats filename3
if (global_min < STATS_min) global_min = STATS_min
if (global_max > STATS_max) global_max = STATS_max

stats filename4
if (global_min < STATS_min) global_min = STATS_min
if (global_max > STATS_max) global_max = STATS_max

if (global_min > 0.005) global_min = global_min - 0.005

plot [][global_min:global_max] filename1 with linespoints linestyle 1 title filename1 noenhanced, \
     filename2 with linespoints linestyle 2 title filename2 noenhanced, \
     filename3 with linespoints linestyle 3 title filename3 noenhanced, \
     filename4 with linespoints linestyle 4 title filename4 noenhanced
